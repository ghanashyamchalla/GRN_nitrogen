409  function nzeroPage = nonZeroWeights(Wfull,lambda,Xsamp,ysamp)
410  /* Return a logical matrix of size numel(lambda)-by-d such that entry
411  *nzero(i,j) = true iff the weight associated with dimension j is
412  *non-zero given l1 regularizer lambda(i). */
413  Wfull = Wfull(end:-1:1,:); /* reverse order for interp1q, (now least regularized to most) */
414 
415  criticalPoints = recoverLambda(Xsamp,ysamp,Wfull)^t; /* in ascending order of magnitude. */
416 
417  tooBig = lambda > criticalPoints(end);/* can't interpolate outside of the range of criticalPoints */
418 
419  Winterp = interp1q(criticalPoints,Wfull,lambda(~tooBig)^t);
420  nzeroPage = Winterp ~=0;
421  nzeroPage = [nzeroPage;false(sum(tooBig),d)]; /* since, if lambda > lambda_max, all weights 0. */
422 
423 
424  /* We have the values of the weights at each 'critical point' where
425  *weights changes sign from lars. Since the weights w.r.t. lambda
426  *are piecewise linear, we can just perform linear interpolation to
427  *get the weights corresponding to lambdas between these points. */
428  end

460 /* % */
461  function lambda = autoLambda(Xs,ys,maxNlambdas,lambda)
462  /* Auto-generate a range of lambdas to search. This is done by
463  * attempting to discover the distribution of lambdas, by performing
464  * lars on (Xs,ys) as well as on resampled data, and then inserting new
465  * values half way between consecutive pairs of existing lambdas until
466  * approximately maxNlambdas have been selected. */
467 
468  if(~strcmpi(lambda," auto "))
469  if(size(lambda,1) > size(lambda,2))
470  lambda = lambda^t; /* make sure its a column vector. */
471 
472  end
473  return;
474  end
475  lambda = recoverLambda(Xs,ys,lars(Xs,ys,larsOptions[:]));
476  k = numel(lambda);
477  while numel(lambda) < ((maxNlambdas/4) - k); /* /4 so that we can double twice via interpolation */
478 
479  [Xsamp,ysamp] = sample(Xs,ys);
480  lambda = [lambda, recoverLambda(Xsamp,ysamp,lars(Xsamp,ysamp,larsOptions[:]))];
481  end
482  lambda = unique(lambda);
483  while numel(lambda) < (maxNlambdas/2)
484  lambda = sort([lambda,filter([0.5,0.5],1,lambda)]); /* filter([0.5,0.5],1,lambda) returns points half way between consecutive lambdas */
485 
486  end
487  lambda = [0,unique(lambda)];
488  end

490 /* % */
491  function lambdaVals = recoverLambda(X,y,Wfull)
492  /* Recover the l1 regularization constants that would result in the
493  * regression weights in Wfull were lasso to be performed on (X,y). Sets
494  * of regression weights, (one per lambda) are stacked row-wise in
495  * Wfull, as returned by lars. */
496  W = Wfull^t;
497  /* dif = bsxfun(@minus,y,X*W); */
498  dif = y * ones(1, size(W, 2)) - X*W;
499  lambdaVals = 2*max(abs(X^t*dif),[],1);
500  end

503  /* Select the final variables by looking for the largest consistent
504  * region on the regularization path, i.e. the largest contiguous region
505  * in which the selected variables do not change. This function is
506  * highly sensitive to the choice of candidate lambdas and will work best
507  * when the lambdas are automatically generated via the autoLambda
508  * function above. */
509 
510  maxcount = 0;
511  lambdaLeftNDX = 0; lambdaRightNDX = 0;
512  nbootstrapsNDX = [];
513  bestVars = [];
514  for b = 1:numel(nbootstraps)
515  currentVars = selectedVars(1,:,b);
516  currentCount = 1;
517  for lam=2:numel(lambda)
518  if(isequal(selectedVars(lam,:,b),currentVars))
519  currentCount = currentCount + 1;
520  else
521  if(currentCount > maxcount)
522  updateCounts;
523  end
524  currentVars = selectedVars(lam,:,b);
525  currentCount = 1;
526  end

528  updateCounts;
529  end
530  end
531  end

539  function updateCounts
540  /* helper sub-function to largestConsistentRegion */

556 /* % */
557  function [bestSelectedVars,bestLambda,bestNbootstraps] = crossValidate(X,y,selectedVars,lambda,nbootstraps,CVnfolds)
558  /* Perform model selection via cross validation. Candidate models
559  *constitute the selected variables corresponding to each value of
560  *lambda and each value of nbootstraps. Models are trained on training
561  *sets via unregularized least squares regression using only the
562  *selected vars corresponding to the the lambda and nbootstrap values
563  *under examination. */
564  allvars = 1:d;
565  [trainfolds, testfolds] = Kfold(n, CVnfolds,1);
566  errors = zeros(numel(nbootstraps),numel(lambda),n);
567  X = [ones(size(X,1),1),X];
568  for f=1:CVnfolds
569  Xtrain = X(trainfolds[f],:); ytrain = y(trainfolds[f],:);
570  Xtest = X(testfolds[f],:); ytest = y(testfolds[f],:);
571  for b = numel(nbootstraps)
572  for lam=1:numel(lambda)
573  vars = [1,allvars(selectedVars(lam,:,b))+1];
574  Wlsq = zeros(d+1,1);
575  Wlsq(vars,1) = Xtrain(:,vars)\ytrain;
576  yhat = Xtest*Wlsq;
577  errors(b,lam,testfolds[f]) = sum((yhat-ytest).^2,2);
578  end
579  end
580  end
581  errMean = mean(errors,3);
582  [val,best] = min(errMean(:));
583  [bsBestNDX,lamBestNDX] = ind2sub(size(errMean),best);
584  bestSelectedVars = allvars(selectedVars(lamBestNDX,:,bsBestNDX));
585  bestLambda = lambda(lamBestNDX);
586  bestNbootstraps = nbootstraps(bsBestNDX);
587  end
